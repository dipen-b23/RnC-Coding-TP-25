#include <stdio.h>
int main()
{
    int n,i,j;
    printf("N = ");
    scanf("%d",&n);
    int arr[n][n];
    printf("Enter array elements: \n");
    for(i=0; i<n; i++)
        for(int j=0; j<n; j++)
            scanf("%d",&arr[i][j]);

    int flag=0;
    for(i=0; i<n; i++)
    {
        for(j=0; j<n; j++)
        {
            if(arr[i][j]!=arr[j][i])
            {
                flag=1;
                break;
            }
        }
    }
    if(flag==0)
      printf("Array is Symmetric");
    else
      printf("Array is Not Symmetric");
    return 0;
}
