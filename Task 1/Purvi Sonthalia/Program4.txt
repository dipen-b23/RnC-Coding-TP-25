#include <stdio.h>
int main()
{
    int m,n,count;
    printf("Enter lower limit: ");
    scanf("%d",&m);
    printf("Enter upper limit: ");
    scanf("%d",&n);
    printf("Prime numbers are: ");
    for(int i=m+1; i<n; i++)
    {
        count=0;
        for(int j=1; j<=i; j++)
        {
            if(i%j==0)
            count++;
        }
        if(count==2)
          printf("%d \t", i);
    }
}
