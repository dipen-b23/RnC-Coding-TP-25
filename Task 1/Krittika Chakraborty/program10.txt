#include <stdio.h>
#include <math.h>
int main()
{
    int n;
    printf("Enter the number:");
    scanf("%d",&n);
    int c=0,n1=n,sum=0;
    while(n1>0)
    {
        c++;
        n1/=10;
    }

    n1=n;
    while(n1>0)
    {
        sum+=pow((n1%10),c);
        n1/=10;
    }
    
    if(sum==n)
    printf("%d is an Armstrong number",n);
    else
    printf("%d is not an Armstrong number",n);
    return 0;
}
