#include <stdio.h>
int main()
{
    int n,i,j;
    printf("Enter the array size:");
    scanf("%d",&n);
    int arr[n];
    printf("Enter the array elements:\n");
    for(i=0; i<n; i++)
        scanf("%d",&arr[i]);
    
    for(i=0; i<n-1; i++)
    {
        printf("enter");
        int min=i;
        for(j=i+1; j<n; j++)
        {
            if(arr[j]<arr[min])
            min=j;
        }
        int t=arr[i];
        arr[i]=arr[min];
        arr[min]=t;
    }
    printf("Sorted Array:");
    for(i=0; i<n; i++)
    printf("%d\t",arr[i]);
    printf("\n");

    printf("Smallest element:%d\n",arr[0]);
    printf("Largest element:%d\n",arr[n-1]);

}
