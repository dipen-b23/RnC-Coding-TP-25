#include <stdio.h>
int main()
{
    int n,s,flag=0;
    printf("Enter the number of elements:");
    scanf("%d",&n);
    int arr[n];
    printf("Enter elements:");
    for(int i=0; i<n; i++)
        scanf("%d",&arr[i]);
    printf("Enter the desired sum:");
    scanf("%d",&s);

    for(int i=0; i<n; i++)
    {
        for(int j=0; j<n; j++)
        {
            if(i==j)
            continue;
            for(int k=0; k<n; k++)
            {
                if(j==k || i==k)
                continue;
                if(s==(arr[i]+arr[j]+arr[k]))
                {
                    printf("Triplets with sum %d: (%d, %d, %d) ",s,arr[i],arr[j],arr[k]);
                    flag=1;
                    break;
                }
                if(flag==1)
                    break;
            }
            if(flag==1)
                break;
        }
    }
    if(flag==0)
    printf("No such triplet found.");
    return 0;
}
