#include <stdio.h>
int main()
{
    int m,n,c;
    printf("Enter the lower limit:");
    scanf("%d",&m);
    printf("Enter the upper limit:");
    scanf("%d",&n);
    printf("The prime numbers are: ");
    for(int i=m+1; i<n; i++)
    {
        c=0;
        for(int j=1; j<=i; j++)
        {
            if(i%j==0)
            c++;
        }
        if(c==2)
        printf("%d \t", i);
    }
}
