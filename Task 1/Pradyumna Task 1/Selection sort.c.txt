#include <stdio.h>
void main(){
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
        
    }
    
    for(int i=0;i<n-1;i++){
        int smallest=arr[i];
        int location=i;
        for(int j=i+1;j<n;j++){
            if(arr[j]< smallest){
                smallest=arr[j];
                location=j;
                
            }
            }
             int temp=arr[i];
            arr[i]=arr[location];
            arr[location]=temp;
    }
    
     for(int i=0;i<n;i++){
        printf("%d",arr[i]);
        
    }
    
}
