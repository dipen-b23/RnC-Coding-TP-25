#include <stdio.h>
#include <math.h>
void main(){
    int num1;
    scanf("%d",&num1);
    int num2=num1;
    int num3=num1;
    int dig=0;
    while(num1!=0){
        num1/=10;
        dig++;
        
    }
    int sum=0;
    while(num2!=0){
        int rem=num2%10;
        sum=sum+pow(rem,dig);
        num2/=10;
        }
    if(num3==sum){
        printf("It is an armstrong number");
    }else{
         printf("It is not an armstrong number");
        
    }
}
