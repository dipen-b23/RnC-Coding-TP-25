#include <stdio.h>

void main(){
int a,b,c;
scanf("%d %d %d",&a,&b,&c);
int h,s1,s2;
if(a>=b&&a>=c){
    h=a;
    s1=b;
    s2=c;
    
}else if(b>=c && b>=a){
    h=b;
    s1=a;
    s2=c;
}else{
    h=c;
    s1=b;
    s2=a;
    
}

if(h*h==s1*s1+s2*s2){
    printf("Triangle is right angled");
    
}else{
    printf("Triangle is not right angled");
    
}

    
}
