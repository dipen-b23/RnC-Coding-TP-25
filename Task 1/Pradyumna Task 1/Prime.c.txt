#include <stdio.h>
void main(){
    int l1,l2;
    scanf("%d %d",&l1,&l2);
    while(l1<=l2){
        int flag=0;
        for(int i=2;i<=l1/2;i++){
            if(l1%i==0){
                flag++;
                break;
            }
            
        }
        if(flag==0){
            printf("%d ,",l1);
            
        }
        l1++;
    }
    
}
