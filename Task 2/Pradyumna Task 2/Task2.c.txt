void avg(int x, int arr[][5]) {
    for (int i = 0; i < x; i++) {
        float sum = 0.0;
        for (int j = 0; j < 5; j++) {
            sum += arr[i][j];
        }
        float average = sum / 5.0f;
        printf("The avg of student %d is = %f \n", i + 1, average);
    }
}

int main() {
    int n;
    printf("Enter number of students: ");
    scanf("%d", &n);

    int arr[n][5];

    for (int i = 0; i < n; i++) {
        printf("Enter 5 marks for student %d: ", i + 1);
        for (int j = 0; j < 5; j++) {
            scanf("%d", &arr[i][j]);
        }
    }

    avg(n, arr);
    return 0;
}

#include <stdio.h>

void sum_rows(int m, int n, int a[10][10])
{
    for (int i = 0; i < m; i++) {
        int sum = 0;
        for (int j = 0; j < n; j++)
            sum += a[i][j];
        printf("Sum of row %d = %d\n", i + 1, sum);
    }
}

void sum_col(int m, int n, int a[10][10])
{
    for (int j = 0; j < n; j++) {
        int sum = 0;
        for (int i = 0; i < m; i++)
            sum += a[i][j];
        printf("Sum of column %d = %d\n", j + 1, sum);
    }
}

void transpose(int m, int n, int a[10][10])
{
    printf("Transpose of the matrix:\n");
    for (int j = 0; j < n; j++) {
        for (int i = 0; i < m; i++)
            printf("%d ", a[i][j]);
        printf("\n");
    }
}

int main(void)
{
    int m, n;
    printf("Enter rows and columns: ");
    scanf("%d %d", &m, &n);

    int a[10][10];  // keep as fixed-size for simplicity
    printf("Enter the matrix elements:\n");
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &a[i][j]);
        }
    }

    int x;
    printf("\nMenu:\n");
    printf("1. Sum of rows\n");
    printf("2. Sum of columns\n");
    printf("3. Transpose\n");
    printf("Enter choice: ");
    scanf("%d", &x);

    switch (x) {
        case 1: sum_rows(m, n, a); break;
        case 2: sum_col(m, n, a);  break;
        case 3: transpose(m, n, a); break;
        default: printf("Invalid choice\n"); break;
    }
    return 0;
}

int sub(char sub1[], char str[])
{
    int pos = 0, flag = -1;
    for (int i = 0; i <= strlen(str) - strlen(sub1); i++)
    {
        pos = i;
        flag = 1;
        for (int j = 0; j < strlen(sub1); j++)
        {
            if (str[i + j] != sub1[j])
            {
                flag = -1;
                break;
            }
        }
        if (flag == 1)
        {
            break;
        }
    }

    return pos;
}
void delete(char del[], char str[])
{
    int pos = sub(del, str);
    for (int i = 0; i <= strlen(del); i++)
    {
        str[i+pos] = str[i +pos+ strlen(del)];
        str[i+pos+ strlen(del)] = '\0';
       
    }
    printf("%s",str);
}
