#include <stdio.h>
int n;
int largest(int i, int arr[n],int max)
{
    if(i==n)
    return max;
    else if(arr[i]>max)
    return largest(++i,arr,arr[i]);
    else
    return largest(++i,arr,max);
}
int main()
{
    printf("Enter the size: ");
    scanf("%d",&n);
    int arr[n];
    printf("Enter elements of array:");
    for(int i=0; i<n; i++)
        scanf("%d",&arr[i]);
    
    int max=largest(0,arr,arr[0]);
    printf("Largest: %d",max);
    return 0;
}
