#include <string.h>

void find(char s1[100])
{

    int flag=0,i;
    printf("Enter substring to be found:");
    scanf(" %[^\n]s", &sub);
    for(i=0;i<strlen(s1);i++)
    {
        strncpy(sub1,s1+i,strlen(sub));
        sub1[strlen(sub)]='\0';
        if(strcmpi(sub,sub1)==0)
        {
            flag=1;
            break;
        }
    }
    if(flag==1)
      printf("substring found\n");
    else
      printf("substring not found\n");
}

int main()
{
    char s[100];
