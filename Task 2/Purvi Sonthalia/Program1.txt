#include <stdio.h>
#include <string.h>
int main()
{
    char str[20];
    printf("Enter the string: ");
    scanf("%s",&str);
    
    int l=strlen(str);
    char str1[l];
    
    for(int i=0;i<l;i++)
        str1[i]=str[l-1-i];
        
    int x=strcmp(str,str1);
    
    if(x==0)
        printf("Palindrome");
    else
        printf("not Palindrome");
        
    return 0;
}
