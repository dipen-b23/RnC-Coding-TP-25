#include <stdio.h>
char* rev(int i, char s[100])
{
    if(i==strlen(s)/2)
    return s;
    else
    {
        char t=s[i];
        s[i]=s[strlen(s)-1-i];
        s[strlen(s)-1-i]=t;
        return rev(++i,s);
    }
}

int main()
{
    char str[100];
    printf("Enter the string:");
    gets(str);
    char *ptr=str;
    ptr=rev(0,str);
    puts(str);
    return 0;
}
