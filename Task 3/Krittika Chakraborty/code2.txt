int potpin=A1;
int ledpin=5;
int potval;
float ledval;
int t=500;
void setup()
{
  pinMode(ledpin, OUTPUT);
  pinMode(potpin,INPUT);
  Serial.begin(9600);
}

void loop()
{
  potval=analogRead(potpin);
  ledval=255.0/1023.0*potval;
  analogWrite(ledpin,ledval);
  Serial.println(ledval);
  delay(t);
}
